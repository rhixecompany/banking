## Split candidates and proposed APIs

1. PlaidLinkProvider

Proposed file: components/layouts/PlaidLinkProvider.tsx

API sketch:

```tsx
"use client";
import Script from "next/script";
export function PlaidLinkProvider({
  children
}: {
  children: React.ReactNode;
}) {
  // Ensures the Plaid script is included once and provides a context for usePlaidLink
  return (
    <>
      <Script
        src="https://cdn.plaid.com/link/v2/stable/link-initialize.js"
        strategy="afterInteractive"
      />
      {children}
    </>
  );
}
```
